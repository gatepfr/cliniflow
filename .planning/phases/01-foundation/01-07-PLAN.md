**`apps/worker/src/index.ts`** — main worker entry:
```typescript
import * as Sentry from '@sentry/node';
import { Worker, Queue } from 'bullmq';
import pino from 'pino';
import { QUEUE_NAMES } from '@clinicaflow/shared';
import { workerConnection, dlqConnection } from './lib/redis.js';
import { processCampaignDispatch } from './queues/campaign-dispatch.js';
import { processAiConversation } from './queues/ai-conversation.js';
import { processAppointmentConfirm } from './queues/appointment-confirm.js';
import { processRecallScheduler } from './queues/recall-scheduler.js';
import { processWebhookEvolution } from './queues/webhook-evolution.js';

if (process.env['SENTRY_DSN']) {
  Sentry.init({ dsn: process.env['SENTRY_DSN'] });
}

const logger = pino({ level: process.env['LOG_LEVEL'] ?? 'info' });

// DLQ for failed jobs — storage only, no processor
const dlq = new Queue(QUEUE_NAMES.DEAD_LETTER, { connection: dlqConnection });

/**
 * Creates a BullMQ worker with retry/DLQ pattern per D-10.
 */
function createWorker<T>(
  queueName: string,
  processor: (job: import('bullmq').Job<T>) => Promise<void>,
  concurrency: number,
): Worker<T> {
  const worker = new Worker<T>(queueName, processor, {
    connection: workerConnection,
    concurrency,
    settings: {
      // D-10: backoff 1s → 5s → 30s (exponential, capped at 30s)
      backoffStrategy: (attemptsMade: number) =>
        Math.min(1000 * Math.pow(2, attemptsMade - 1), 30_000),
    },
    defaultJobOptions: {
      attempts: 3,
      backoff: { type: 'custom' },
      removeOnComplete: { age: 24 * 3600 },
      removeOnFail: false, // keep in failed set until DLQ moves it
    },
  });

  worker.on('failed', async (job, err) => {
    if (!job) return;
    const maxAttempts = job.opts.attempts ?? 3;
    if (job.attemptsMade >= maxAttempts) {
      // Move to DLQ (D-10)
      await dlq.add(`dlq_${queueName}`, {
        originalQueue: queueName,
        jobId: job.id,
        jobName: job.name,
        data: job.data,
        error: err.message,
        attemptsMade: job.attemptsMade,
        failedAt: new Date().toISOString(),
        // stacktrace intentionally omitted — may contain path info
      });

      // Log to Sentry without PII
      Sentry.captureException(err, {
        extra: {
          jobId: job.id,
          queue: queueName,
          attemptsMade: job.attemptsMade,
        },
      });

      logger.error(
        { jobId: job.id, queue: queueName, err: err.message },
        'Job moved to DLQ after max retries'
      );
    }
  });

  worker.on('error', (err) => {
    logger.error({ err: err.message }, `Worker error on queue ${queueName}`);
  });

  logger.info({ queue: queueName, concurrency }, 'Worker started');
  return worker;
}

// Start all 5 workers — D-08 (single process), D-09 (concurrency per queue)
createWorker(QUEUE_NAMES.CAMPAIGN_DISPATCH,   processCampaignDispatch,   5);
createWorker(QUEUE_NAMES.AI_CONVERSATION,     processAiConversation,     10);
createWorker(QUEUE_NAMES.APPOINTMENT_CONFIRM, processAppointmentConfirm, 3);
createWorker(QUEUE_NAMES.RECALL_SCHEDULER,    processRecallScheduler,    1);
createWorker(QUEUE_NAMES.WEBHOOK_EVOLUTION,   processWebhookEvolution,   20);

logger.info('All 5 BullMQ workers started');

// Graceful shutdown
async function shutdown() {
  logger.info('Shutting down workers...');
  // Workers close themselves; DLQ queue needs explicit close
  await dlq.close();
  await dlqConnection.quit();
  await workerConnection.quit();
  process.exit(0);
}

process.on('SIGTERM', () => void shutdown());
process.on('SIGINT', () => void shutdown());
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| BullMQ job data | Jobs are enqueued by API or scheduler; data must not contain PII per LGPD |
| Sentry error reporting | Error metadata sent to Sentry; must exclude PII (patient data) |
| DLQ | Failed jobs stored in Redis for inspection; job.data must be auditable without PII exposure |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-1-PLAN07-01 | Information Disclosure | PII in DLQ job data | mitigate | Job data types (`CampaignDispatchJobData`, etc.) contain only IDs (tenantId, patientId, campaignId) — never full_name, phone_normalized, or content. Processor stubs enforce this contract. |
| T-1-PLAN07-02 | Information Disclosure | PII in Sentry error payload | mitigate | `Sentry.captureException()` extra payload contains only `jobId`, `queue`, `attemptsMade` — no patient data fields. |
| T-1-PLAN07-03 | Denial of Service | Worker crash on Redis reconnect | mitigate | `maxRetriesPerRequest: null` on `workerConnection` prevents MaxRetriesPerRequestError from killing the process on transient Redis unavailability. |
  | T-1-PLAN07-04 | Tampering | Job data injection via enqueue API | accept | Phase 1 workers are stubs; full validation added per-queue in Phases 3-5 when actual processing is implemented. |
</threat_model>
