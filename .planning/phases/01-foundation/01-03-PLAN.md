**`packages/db/src/client.ts`** — Extended Prisma client:

CRITICAL: Use `baseClient` (NOT the extended `prisma`) for audit log writes. The `$extends` intercepts ALL operations on the extended client, including writes to AuditLog, which would cause infinite recursion (Pitfall 5 from RESEARCH.md).

CRITICAL: Models not in TENANT_SCOPED_MODELS (AuditLog, Tenant, User) must be skipped entirely — do NOT call `getTenantContext()` for them because they either have no `tenantId` or are queried before a context exists.

```typescript
import { PrismaClient } from '@prisma/client';
import { getTenantContext } from './context.js';

// Models that require tenant isolation (Prisma uses PascalCase for model names in $extends)
const TENANT_SCOPED_MODELS = new Set([
  'Patient', 'Visit', 'Treatment', 'Campaign', 'Message',
  'Conversation', 'ChatMessage', 'Appointment', 'AiConfig',
] as const);

// Models where mutations generate audit log entries
const AUDITED_MODELS = new Set([
  'Patient', 'Visit', 'Treatment', 'Conversation', 'ChatMessage', 'Appointment',
] as const);

const WRITE_OPERATIONS = new Set([
  'create', 'update', 'delete', 'upsert',
  'createMany', 'updateMany', 'deleteMany',
] as const);

const READ_OPERATIONS = new Set([
  'findUnique', 'findFirst', 'findMany', 'count', 'aggregate', 'groupBy',
] as const);

// Base client WITHOUT extends — used for audit log writes to prevent recursion
export const baseClient = new PrismaClient({
  log: process.env.NODE_ENV === 'development' ? ['error', 'warn'] : ['error'],
});

export const prisma = baseClient.$extends({
  query: {
    $allModels: {
      async $allOperations({ model, operation, args, query }: {
        model: string | undefined;
        operation: string;
        args: Record<string, unknown>;
        query: (args: Record<string, unknown>) => Promise<unknown>;
      }) {
        // Skip non-tenant-scoped models (AuditLog, Tenant, User, and undefined)
        if (!model || !TENANT_SCOPED_MODELS.has(model as never)) {
          return query(args);
        }

        // Fail-fast: require tenant context for ALL scoped operations
        const ctx = getTenantContext(); // throws [SECURITY] error if no ALS context

        // Inject tenantId into WHERE clause for reads and writes
        if (READ_OPERATIONS.has(operation as never)) {
          args = { ...args, where: { ...(args.where as object ?? {}), tenantId: ctx.tenantId } };
        }

        if (WRITE_OPERATIONS.has(operation as never)) {
          // Scope the WHERE clause to tenant
          args = { ...args, where: { ...(args.where as object ?? {}), tenantId: ctx.tenantId } };
          // Inject tenantId into data for create/upsert
          if (operation === 'create' || operation === 'createMany') {
            args = { ...args, data: { ...(args.data as object ?? {}), tenantId: ctx.tenantId } };
          }
          if (operation === 'upsert') {
            args = {
              ...args,
              create: { ...(args.create as object ?? {}), tenantId: ctx.tenantId },
              update: args.update,
            };
          }
        }

        const result = await query(args);

        // Write audit log for mutations on audited models
        // MUST use baseClient — NOT prisma — to prevent recursion
        if (
          AUDITED_MODELS.has(model as never) &&
          WRITE_OPERATIONS.has(operation as never)
        ) {
          const entityId = (result as { id?: string } | null)?.id ?? 'batch';
          await baseClient.auditLog.create({
            data: {
              id: (await import('@paralleldrive/cuid2')).createId(),
              tenantId: ctx.tenantId,
              userId: ctx.userId,
              action: operation,
              entity: model,
              entityId,
              metadata: {}, // NEVER put PII here — LGPD art. 11
            },
          });
        }

        return result;
      },
    },
  },
});

export type PrismaExtended = typeof prisma;
```
